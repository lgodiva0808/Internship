from selenium.webdriver.common.by import By
from time import sleep

from pages.base_page import Page

class CommunityPage(Page):

    CONTACT_SUPPORT_BTN = (By.CSS_SELECTOR, '#w-node-_746955f8-6d08-524d-7f43-03daaced30ff-7f66deba')


    def verify_community_page_opens(self):
        sleep(3)
        self.wait_until_url_contains('community')


    def verify_contact_support_button(self):
        sleep(2)
        self.wait_until_clickable_click(*self.CONTACT_SUPPORT_BTN)
